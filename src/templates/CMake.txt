cmake_minimum_required(VERSION 3.0)

project(parallel-hnsw)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

if(CMAKE_CXX_COMPILER_ID MATCHES "MSVC")
	set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS}@ARCH@")
endif()

file(GLOB headers chm/*.hpp)
file(GLOB sources chm/*.cpp)
source_group("Headers" FILES ${headers})
source_group("Sources" FILES ${sources})

add_library(chmLib ${headers} ${sources})@LIB_DEFS@
target_include_directories(chmLib PUBLIC "${CMAKE_CURRENT_SOURCE_DIR}")

add_executable(benchmark executables/benchmark.cpp)@EXE_DEFS@
target_include_directories(benchmark PUBLIC "${PROJECT_BINARY_DIR}" "${PROJECT_SOURCE_DIR}")
target_link_libraries(benchmark PUBLIC chmLib)
